#include <stdio.h>
#include <math.h>
int Function(FILE*vvod , char nazvanie[1000])
{
  int current;
  int previous;
  int counter = 0;
  vvod = fopen(nazvanie , "r");
  if(fscanf(vvod , "%d" , &current) == EOF)
  return-1;
  counter ++;
  previous = current;
  while(!feof(vvod))
  {
   fscanf(vvod , "%d" , &current);
   if(previous > current)
    {
    counter = -2;
    break;
    } 
   if(previous < current)
    {
    counter ++;
    previous = current;
    }
  }
  fclose(vvod);
return(counter);
}

int main(void)
{
  FILE*vvod;
  char nazvanie[1000];
  printf("Введите название файла: ");
  scanf("%s" , nazvanie);
  vvod = fopen(nazvanie , "r");
  if(vvod == NULL)
   {
    printf("Не удалось открыть файл");
    return -1;
   }
  fclose(vvod); 
   if(Function(vvod , nazvanie) == -1)
   printf("Файл пустой");
   if(Function(vvod , nazvanie) == -2)
   printf("Последовательность не является неубывающей");
   if(Function(vvod , nazvanie) > 0)
   printf("%d", Function(vvod , nazvanie));
return 0;
}
